


# Basic test for searching styles

#
Lang it_IT.utf8
TestBegin test.lyx -dbg find > lyx-log.txt 2>&1
KK: foo\[Return]
KK: \Ap2
KK: foo\[Return]
KK: \Aps
KK: foo\C\[Home]
KK: \Afs
KK: \CF
# Uncheck ignore format
KK: \Az\Ag\Ae
KK: \Ap2
KK: foo\[Return]
TestEnd
Assert pcregrep -M 'Putting selection at .*idx: 0 par: 1 pos: 0\n with len: 3' lyx-log.txt
#

KK: \C\[Home]
KK: \CF
# Uncheck ignore format
KK: \Az\Ag\Ae
KK: foo\[Return]\[Return]
TestEnd
